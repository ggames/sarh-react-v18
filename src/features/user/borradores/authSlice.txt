// src/features/auth/authSlice.ts
import { createSlice } from "@reduxjs/toolkit";
import { loginUser } from "./authThunk";

interface AuthState {
  user: string | null;
  token: string | null; // persistente (localStorage)
  tempToken: string | null; // temporal (memoria)
  roles: string[];
  loading: boolean;
  error: string | null;
  isAuthenticated: boolean;
}

const initialState: AuthState = {
  user: localStorage.getItem("user"),

  token: localStorage.getItem("token"), // si hay token guardado, lo cargamos
  tempToken: null,
  roles: JSON.parse(localStorage.getItem("roles_token") || "[]")    ,
  loading: false,
  error: null,
  isAuthenticated: false,
};

const authSlice = createSlice({
  name: "auth",
  initialState,
  reducers: {
    logout(state) {
      state.user = null;

      state.token = null;
      state.tempToken = null;
      state.roles = [];
      state.error = null;
      state.isAuthenticated = false;
      localStorage.clear();
    },
    setToken(state, action) {
      state.token = action.payload;

      localStorage.setItem("token", action.payload);
    }
 
  },
  extraReducers: (builder) => {
    builder
      // loginUser.pending
      .addCase(loginUser.pending, (state) => {
        state.loading = true;
        state.error = null;
      })
      // loginUser.fulfilled
      .addCase(loginUser.fulfilled, (state, action) => {
        state.loading = false;
        state.token = action.payload.token ?? null;
        state.tempToken = action.payload.tempToken ?? null; // opcional

        state.user = action.payload.user;

        state.roles = action.payload.roles ?? [];
         // persistente
        
    
      })
      // loginUser.rejected
      .addCase(loginUser.rejected, (state, action) => {
        state.loading = false;
        state.error = action.payload as string;
      });
  },
});

export const { logout, setToken } =
  authSlice.actions;
export default authSlice.reducer;
